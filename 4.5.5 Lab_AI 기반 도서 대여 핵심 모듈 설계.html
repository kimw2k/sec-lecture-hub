class LibraryService:
    def process_rental(self, member: Member, book: Book):
        # 1. 연체 여부 확인 (Overdue Check)
        if member.overdue_days > 0:
            return {"status": "FAILED", "reason": "OVERDUE"}

        # 2. 등급별 대여 한도 도달 확인
        max_limit = 5 if member.grade == "Premium" else 3
        if len(member.rented_books) >= max_limit:
            return {"status": "FAILED", "reason": "LIMIT_EXCEEDED"}

        # 3. 대여 처리
        member.rented_books.append(book)
        return {"status": "SUCCESS"}
